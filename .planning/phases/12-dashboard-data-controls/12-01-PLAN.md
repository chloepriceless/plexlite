Create `dvhub/public/components/dashboard/dashboard-compute.js` with these pure functions ported from old app.js:

1. **`formatCentFromCt(ct)`** -- Port from app.js line 20-22. Takes cents value, returns formatted string using de-DE locale with "Cent" suffix. Returns '-' for non-finite input. Example: `formatCentFromCt(12.345)` -> `"12,35 Cent"`.

2. **`formatCentFromTenthCt(value)`** -- Port from app.js line 24-26. Divides by 10 first, then calls formatCentFromCt logic. Example: `formatCentFromTenthCt(123)` -> `"12,3 Cent"`.

3. **`resolveDvControlIndicators(status)`** -- Port from app.js lines 696-727. Takes a status object (the telemetry.value), returns `{ dc: { text, tone }, ac: { text, tone } }`. Logic:
   - First checks `status.victron.feedExcessDcPv` and `status.victron.dontFeedExcessAcPv` (legacy Victron readback).
   - DC: value 1 = text 'EIN' tone 'ok', value 0 = text 'AUS' tone 'off', null = text '-'.
   - AC: value 1 = text 'Ja' tone 'off', value 0 = text 'Nein' tone 'ok', null = text '-'.
   - Falls back to `status.ctrl.dvControl` if victron fields are both null.
   - With dvControl: uses `dvc.feedIn` boolean. feedIn true: DC='EIN'/ok, AC='Nein'/ok. feedIn false: DC='AUS'/off, AC='Ja'/off.

4. **`computeCostColor(netEur)`** -- Returns CSS variable string. `netEur >= 0` -> `'var(--dvhub-green)'`, else `'var(--dvhub-red)'`. Returns `'var(--text-muted)'` for null/undefined.

5. **`formatTimestamp(ts)`** -- Takes ISO string or epoch ms, returns `"DD.MM. HH:MM"` German locale format. Returns `'--'` for falsy input.

All functions use `export function` (ES module exports). No DOM, no signals, no imports from other project files.

import { html } from 'htm/preact';

        <span>Aktuell</span>

      </div>
      <div class="metric-row">
        <span>Naechster Slot</span>
        <strong>${formatCentFromCt(epex.next)}</strong>
      </div>
      <div class="metric-row">
        <span>Heute Min</span>
        <strong style="color:var(--dvhub-green)">${formatCentFromTenthCt(epex.todayMin)}</strong>

**Create `dvhub/public/components/dashboard/status-card.js`:**
```javascript
import { html } from 'htm/preact';
import { telemetry } from '../shared/use-signal-store.js';
import { formatPower } from '../shared/format.js';
import { resolveDvControlIndicators, formatTimestamp } from './dashboard-compute.js';

export function StatusCard() {
  const t = telemetry.value || {};
  const meter = t.meter || {};
  const dv = resolveDvControlIndicators(t);
  const negPrice = t.ctrl?.negativePriceActive;
  const keepaliveTs = t.keepalive?.modbusLastQuery?.ts;
  return html`
    <section class="panel span-4 reveal">
      <p class="card-title">System-Status</p>
      <div class="metric-row">
        <span>L1</span><strong>${formatPower(meter.grid_l1_w)}</strong>
      </div>
      <div class="metric-row">
        <span>L2</span><strong>${formatPower(meter.grid_l2_w)}</strong>
      </div>
      <div class="metric-row">
        <span>L3</span><strong>${formatPower(meter.grid_l3_w)}</strong>
      </div>
      <div class="metric-row">
        <span>DV DC</span>
        <strong style="color:${dv.dc.tone === 'ok' ? 'var(--dvhub-green)' : dv.dc.tone === 'off' ? 'var(--dvhub-red)' : 'var(--text-muted)'}">${dv.dc.text}</strong>
      </div>
      <div class="metric-row">
        <span>DV AC</span>
        <strong style="color:${dv.ac.tone === 'ok' ? 'var(--dvhub-green)' : dv.ac.tone === 'off' ? 'var(--dvhub-red)' : 'var(--text-muted)'}">${dv.ac.text}</strong>
      </div>
      <div class="metric-row">
        <span>Neg. Preis</span>
        <strong style="color:${negPrice ? 'var(--dvhub-orange)' : 'var(--text-muted)'}">${negPrice ? 'AKTIV' : 'Inaktiv'}</strong>
      </div>
      <div class="metric-row">
        <span>Keepalive</span>
        <strong style="color:var(--text-muted)">${formatTimestamp(keepaliveTs)}</strong>
      </div>
    </section>
  `;
}
```
