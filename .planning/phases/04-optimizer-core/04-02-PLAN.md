    1. import { BehaviorSubject } from 'rxjs'

    3. export function createPlanEngine({ scorer, maxHistory = 50 } = {}):
       - const activePlan$ = new BehaviorSubject(null)
       - const history = []  // Array of { plan, score, receivedAt }
       - return { submitPlan, getActivePlan, getActivePlan$, getHistory, clearActivePlan, destroy }

       submitPlan(plan):
       - const score = scorer.scorePlan(plan)
       - const entry = { plan, score, receivedAt: new Date().toISOString() }
       - history.unshift(entry)  // newest first
       - if (history.length > maxHistory) history.length = maxHistory
       - Re-evaluate winner: call chooseWinningPlan on all feasible entries in history
       - const result = chooseWinningPlan(history.filter(h => h.score.feasible !== false).map(h => ({ ...h.plan, score: h.score })))
       - if (result.active): activePlan$.next(result.active)
       - else: activePlan$.next(null)
       - return { entry, isNewWinner: result.active?.runId === plan.runId }

       getActivePlan(): return activePlan$.getValue()

       getActivePlan$(): return activePlan$.asObservable()

       getHistory({ limit } = {}): return limit ? history.slice(0, limit) : [...history]

       clearActivePlan(): activePlan$.next(null)

       destroy(): activePlan$.complete()
